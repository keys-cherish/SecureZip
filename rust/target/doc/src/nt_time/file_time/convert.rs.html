15impl From<FileTime> for u64 {
16    /// Converts a `FileTime` to the file time.
17    ///
18    /// Equivalent to [`FileTime::to_raw`] except that it is not callable in
19    /// const contexts.
20    ///
21    /// # Examples
22    ///
23    /// ```
24    /// # use nt_time::FileTime;
25    /// #
26    /// assert_eq!(u64::from(FileTime::NT_TIME_EPOCH), u64::MIN);
27    /// assert_eq!(u64::from(FileTime::UNIX_EPOCH), 116_444_736_000_000_000);
28    /// assert_eq!(u64::from(FileTime::MAX), u64::MAX);
29    /// ```
30    #[inline]
31    fn from(ft: FileTime) -> Self {
32        ft.to_raw()
33    }
34}

36impl TryFrom<FileTime> for i64 {
37    type Error = TryFromIntError;
38
39    /// Converts a `FileTime` to the file time.
40    ///
41    /// The file time may be represented as an [`i64`] value in [WinRT],[^clock]
42    /// [.NET],[^fromfiletime][^tofiletime] etc.
43    ///
44    /// # Errors
45    ///
46    /// Returns [`Err`] if `ft` is after "+30828-09-14 02:48:05.477580700 UTC".
47    ///
48    /// # Examples
49    ///
50    /// ```
51    /// # use nt_time::FileTime;
52    /// #
53    /// assert_eq!(
54    ///     i64::try_from(FileTime::NT_TIME_EPOCH).unwrap(),
55    ///     i64::default()
56    /// );
57    /// assert_eq!(
58    ///     i64::try_from(FileTime::UNIX_EPOCH).unwrap(),
59    ///     116_444_736_000_000_000
60    /// );
61    ///
62    /// assert!(i64::try_from(FileTime::MAX).is_err());
63    /// ```
64    ///
65    /// [^clock]: <https://learn.microsoft.com/en-us/uwp/cpp-ref-for-winrt/clock>
66    ///
67    /// [^fromfiletime]: <https://learn.microsoft.com/en-us/dotnet/api/system.datetime.fromfiletime>
68    ///
69    /// [^tofiletime]: <https://learn.microsoft.com/en-us/dotnet/api/system.datetime.tofiletime>
70    ///
71    /// [WinRT]: https://learn.microsoft.com/en-us/windows/uwp/cpp-and-winrt-apis/
72    /// [.NET]: https://dotnet.microsoft.com/
73    #[inline]
74    fn try_from(ft: FileTime) -> Result<Self, Self::Error> {
75        ft.to_raw().try_into()
76    }
77}

79#[cfg(feature = "std")]
80impl From<FileTime> for std::time::SystemTime {
81    /// Converts a `FileTime` to a [`SystemTime`](std::time::SystemTime).
82    ///
83    /// # Panics
84    ///
85    /// Panics if the resulting time cannot be represented by a
86    /// [`SystemTime`](std::time::SystemTime).
87    ///
88    /// # Examples
89    ///
90    /// ```
91    /// # use std::time::{Duration, SystemTime};
92    /// #
93    /// # use nt_time::FileTime;
94    /// #
95    /// assert_eq!(
96    ///     SystemTime::from(FileTime::NT_TIME_EPOCH),
97    ///     SystemTime::UNIX_EPOCH - Duration::from_secs(11_644_473_600)
98    /// );
99    /// assert_eq!(
100    ///     SystemTime::from(FileTime::UNIX_EPOCH),
101    ///     SystemTime::UNIX_EPOCH
102    /// );
103    /// ```
104    fn from(ft: FileTime) -> Self {
105        use std::time::Duration;
106
107        use super::FILE_TIMES_PER_SEC;
108
109        let duration = Duration::new(
110            ft.to_raw() / FILE_TIMES_PER_SEC,
111            u32::try_from((ft.to_raw() % FILE_TIMES_PER_SEC) * 100)
112                .expect("the number of nanoseconds should be in the range of `u32`"),
113        );
114        (Self::UNIX_EPOCH - (FileTime::UNIX_EPOCH - FileTime::NT_TIME_EPOCH)) + duration
115    }
116}

118impl TryFrom<FileTime> for OffsetDateTime {
119    type Error = ComponentRange;
120
121    /// Converts a `FileTime` to a [`OffsetDateTime`].
122    ///
123    /// # Errors
124    ///
125    /// Returns [`Err`] if `time` is out of range for [`OffsetDateTime`].
126    ///
127    /// # Examples
128    ///
129    /// ```
130    /// # use nt_time::{
131    /// #     time::{macros::datetime, OffsetDateTime},
132    /// #     FileTime,
133    /// # };
134    /// #
135    /// assert_eq!(
136    ///     OffsetDateTime::try_from(FileTime::NT_TIME_EPOCH).unwrap(),
137    ///     datetime!(1601-01-01 00:00 UTC)
138    /// );
139    /// assert_eq!(
140    ///     OffsetDateTime::try_from(FileTime::UNIX_EPOCH).unwrap(),
141    ///     OffsetDateTime::UNIX_EPOCH
142    /// );
143    /// ```
144    ///
145    /// With the `large-dates` feature disabled, returns [`Err`] if the file
146    /// time represents after "9999-12-31 23:59:59.999999900 UTC":
147    ///
148    /// ```
149    /// # use nt_time::{time::OffsetDateTime, FileTime};
150    /// #
151    /// # #[cfg(not(feature = "large-dates"))]
152    /// assert!(OffsetDateTime::try_from(FileTime::new(2_650_467_744_000_000_000)).is_err());
153    /// ```
154    ///
155    /// With the `large-dates` feature enabled, this always succeeds:
156    ///
157    /// ```
158    /// # use nt_time::{
159    /// #     time::{macros::datetime, OffsetDateTime},
160    /// #     FileTime,
161    /// # };
162    /// #
163    /// # #[cfg(feature = "large-dates")]
164    /// assert_eq!(
165    ///     OffsetDateTime::try_from(FileTime::new(2_650_467_744_000_000_000)).unwrap(),
166    ///     datetime!(+10000-01-01 00:00 UTC)
167    /// );
168    /// # #[cfg(feature = "large-dates")]
169    /// assert_eq!(
170    ///     OffsetDateTime::try_from(FileTime::MAX).unwrap(),
171    ///     datetime!(+60056-05-28 05:36:10.955_161_500 UTC)
172    /// );
173    /// ```
174    fn try_from(ft: FileTime) -> Result<Self, Self::Error> {
175        Self::from_unix_timestamp_nanos(ft.to_unix_time_nanos())
176    }
177}

179#[cfg(feature = "chrono")]
180impl From<FileTime> for chrono::DateTime<chrono::Utc> {
181    /// Converts a `FileTime` to a [`DateTime<Utc>`](chrono::DateTime).
182    ///
183    /// # Examples
184    ///
185    /// ```
186    /// # use nt_time::{
187    /// #     chrono::{DateTime, Utc},
188    /// #     FileTime,
189    /// # };
190    /// #
191    /// assert_eq!(
192    ///     DateTime::<Utc>::from(FileTime::NT_TIME_EPOCH),
193    ///     "1601-01-01 00:00:00 UTC".parse::<DateTime<Utc>>().unwrap()
194    /// );
195    /// assert_eq!(
196    ///     DateTime::<Utc>::from(FileTime::UNIX_EPOCH),
197    ///     DateTime::<Utc>::UNIX_EPOCH
198    /// );
199    /// ```
200    fn from(ft: FileTime) -> Self {
201        let ut = ft.to_unix_time_nanos();
202        Self::from_timestamp(
203            i64::try_from(ut / 1_000_000_000)
204                .expect("the number of seconds should be in the range of `i64`"),
205            u32::try_from(ut % 1_000_000_000)
206                .expect("the number of nanoseconds should be in the range of `u32`"),
207        )
208        .expect("Unix time in nanoseconds should be in the range of `DateTime<Utc>`")
209    }
210}

212#[cfg(feature = "zip")]
213impl TryFrom<FileTime> for zip::DateTime {
214    type Error = zip::result::DateTimeRangeError;
215
216    /// Converts a `FileTime` to a [`zip::DateTime`].
217    ///
218    /// This method returns the UTC date and time. The resolution of
219    /// [`zip::DateTime`] is 2 seconds, so the part of `ft` which is less than 2
220    /// seconds is truncated.
221    ///
222    /// # Errors
223    ///
224    /// Returns [`Err`] if `ft` is out of range for [`zip::DateTime`].
225    ///
226    /// # Examples
227    ///
228    /// ```
229    /// # use nt_time::{zip::DateTime, FileTime};
230    /// #
231    /// # {
232    /// // `1980-01-01 00:00:00 UTC`.
233    /// let dt = DateTime::try_from(FileTime::new(119_600_064_000_000_000)).unwrap();
234    /// assert_eq!(dt.year(), 1980);
235    /// assert_eq!(dt.month(), 1);
236    /// assert_eq!(dt.day(), 1);
237    /// assert_eq!(dt.hour(), 0);
238    /// assert_eq!(dt.minute(), 0);
239    /// assert_eq!(dt.second(), 0);
240    /// # }
241    /// # {
242    /// // `2107-12-31 23:59:58 UTC`.
243    /// let dt = DateTime::try_from(FileTime::new(159_992_927_980_000_000)).unwrap();
244    /// assert_eq!(dt.year(), 2107);
245    /// assert_eq!(dt.month(), 12);
246    /// assert_eq!(dt.day(), 31);
247    /// assert_eq!(dt.hour(), 23);
248    /// assert_eq!(dt.minute(), 59);
249    /// assert_eq!(dt.second(), 58);
250    /// # }
251    ///
252    /// # {
253    /// // `2107-12-31 23:59:59 UTC`.
254    /// let dt = DateTime::try_from(FileTime::new(159_992_927_990_000_000)).unwrap();
255    /// assert_eq!(dt.year(), 2107);
256    /// assert_eq!(dt.month(), 12);
257    /// assert_eq!(dt.day(), 31);
258    /// assert_eq!(dt.hour(), 23);
259    /// assert_eq!(dt.minute(), 59);
260    /// assert_eq!(dt.second(), 58);
261    /// # }
262    ///
263    /// // Before `1980-01-01 00:00:00`.
264    /// assert!(DateTime::try_from(FileTime::new(119_600_063_990_000_000)).is_err());
265    /// // After `2107-12-31 23:59:58`.
266    /// assert!(DateTime::try_from(FileTime::new(159_992_928_000_000_000)).is_err());
267    /// ```
268    fn try_from(ft: FileTime) -> Result<Self, Self::Error> {
269        use zip::result::DateTimeRangeError;
270
271        ft.to_dos_date_time(None)
272            .map_err(|_| DateTimeRangeError)
273            .and_then(|(date, time, ..)| Self::try_from_msdos(date, time))
274    }
275}

301impl TryFrom<i64> for FileTime {
302    type Error = FileTimeRangeError;
303
304    /// Converts the file time to a `FileTime`.
305    ///
306    /// The file time may be represented as an [`i64`] value in [WinRT],[^clock]
307    /// [.NET],[^fromfiletime][^tofiletime] etc.
308    ///
309    /// # Errors
310    ///
311    /// Returns [`Err`] if `ft` is negative.
312    ///
313    /// # Examples
314    ///
315    /// ```
316    /// # use nt_time::FileTime;
317    /// #
318    /// assert_eq!(
319    ///     FileTime::try_from(i64::default()).unwrap(),
320    ///     FileTime::NT_TIME_EPOCH
321    /// );
322    /// assert_eq!(
323    ///     FileTime::try_from(116_444_736_000_000_000_i64).unwrap(),
324    ///     FileTime::UNIX_EPOCH
325    /// );
326    ///
327    /// assert!(FileTime::try_from(i64::MIN).is_err());
328    /// ```
329    ///
330    /// [^clock]: <https://learn.microsoft.com/en-us/uwp/cpp-ref-for-winrt/clock>
331    ///
332    /// [^fromfiletime]: <https://learn.microsoft.com/en-us/dotnet/api/system.datetime.fromfiletime>
333    ///
334    /// [^tofiletime]: <https://learn.microsoft.com/en-us/dotnet/api/system.datetime.tofiletime>
335    ///
336    /// [WinRT]: https://learn.microsoft.com/en-us/windows/uwp/cpp-and-winrt-apis/
337    /// [.NET]: https://dotnet.microsoft.com/
338    #[inline]
339    fn try_from(ft: i64) -> Result<Self, Self::Error> {
340        ft.try_into()
341            .map_err(|_| FileTimeRangeErrorKind::Negative.into())
342            .map(Self::new)
343    }
344}

346#[cfg(feature = "std")]
347impl TryFrom<std::time::SystemTime> for FileTime {
348    type Error = FileTimeRangeError;
349
350    /// Converts a [`SystemTime`](std::time::SystemTime) to a `FileTime`.
351    ///
352    /// # Errors
353    ///
354    /// Returns [`Err`] if `time` is out of range for the file time.
355    ///
356    /// # Examples
357    ///
358    /// ```
359    /// # use std::time::{Duration, SystemTime};
360    /// #
361    /// # use nt_time::FileTime;
362    /// #
363    /// assert_eq!(
364    ///     FileTime::try_from(SystemTime::UNIX_EPOCH - Duration::from_secs(11_644_473_600)).unwrap(),
365    ///     FileTime::NT_TIME_EPOCH
366    /// );
367    /// assert_eq!(
368    ///     FileTime::try_from(SystemTime::UNIX_EPOCH).unwrap(),
369    ///     FileTime::UNIX_EPOCH
370    /// );
371    ///
372    /// // Before `1601-01-01 00:00:00 UTC`.
373    /// assert!(FileTime::try_from(
374    ///     SystemTime::UNIX_EPOCH - Duration::from_nanos(11_644_473_600_000_000_100)
375    /// )
376    /// .is_err());
377    /// // After `+60056-05-28 05:36:10.955161500 UTC`.
378    /// #[cfg(not(windows))]
379    /// assert!(FileTime::try_from(
380    ///     SystemTime::UNIX_EPOCH + Duration::new(1_833_029_933_770, 955_161_600)
381    /// )
382    /// .is_err());
383    /// ```
384    fn try_from(st: std::time::SystemTime) -> Result<Self, Self::Error> {
385        use std::time::SystemTime;
386
387        let elapsed = st
388            .duration_since(SystemTime::UNIX_EPOCH - (Self::UNIX_EPOCH - Self::NT_TIME_EPOCH))
389            .map(|d| d.as_nanos())
390            .map_err(|_| FileTimeRangeErrorKind::Negative)?;
391        let ft = u64::try_from(elapsed / 100).map_err(|_| FileTimeRangeErrorKind::Overflow)?;
392        Ok(Self::new(ft))
393    }
394}

452
453    /// Converts a [`DateTime<Utc>`](chrono::DateTime) to a `FileTime`.
454    ///
455    /// # Errors
456    ///
457    /// Returns [`Err`] if `dt` is out of range for the file time.
458    ///
459    /// # Examples
460    ///
461    /// ```
462    /// # use nt_time::{
463    /// #     chrono::{DateTime, TimeDelta, Utc},
464    /// #     FileTime,
465    /// # };
466    /// #
467    /// assert_eq!(
468    ///     FileTime::try_from("1601-01-01 00:00:00 UTC".parse::<DateTime<Utc>>().unwrap()).unwrap(),
469    ///     FileTime::NT_TIME_EPOCH
470    /// );
471    /// assert_eq!(
472    ///     FileTime::try_from(DateTime::<Utc>::UNIX_EPOCH).unwrap(),
473    ///     FileTime::UNIX_EPOCH
474    /// );
475    ///
476    /// // Before `1601-01-01 00:00:00 UTC`.
477    /// assert!(FileTime::try_from(
478    ///     "1601-01-01 00:00:00 UTC".parse::<DateTime<Utc>>().unwrap() - TimeDelta::nanoseconds(100)
479    /// )
480    /// .is_err());
481    /// // After `+60056-05-28 05:36:10.955161500 UTC`.
482    /// assert!(FileTime::try_from(
483    ///     "+60056-05-28 05:36:10.955161500 UTC"
484    ///         .parse::<DateTime<Utc>>()
485    ///         .unwrap()
486    ///         + TimeDelta::nanoseconds(100)
487    /// )
488    /// .is_err());
489    /// ```
490    fn try_from(dt: chrono::DateTime<chrono::Utc>) -> Result<Self, Self::Error> {
491        let ut =
492            (i128::from(dt.timestamp()) * 1_000_000_000) + i128::from(dt.timestamp_subsec_nanos());
493        Self::from_unix_time_nanos(ut)
494    }
495}

497#[cfg(feature = "zip")]
498impl From<zip::DateTime> for FileTime {
499    /// Converts a [`zip::DateTime`] to a `FileTime`.
500    ///
501    /// This method assumes `dt` is in UTC.
502    ///
503    /// # Panics
504    ///
505    /// Panics if `dt` is an invalid [`zip::DateTime`].
506    ///
507    /// # Examples
508    ///
509    /// ```
510    /// # use nt_time::{zip::DateTime, FileTime};
511    /// #
512    /// assert_eq!(
513    ///     FileTime::from(DateTime::from_date_and_time(1980, 1, 1, 0, 0, 0).unwrap()),
514    ///     FileTime::new(119_600_064_000_000_000)
515    /// );
516    /// assert_eq!(
517    ///     FileTime::from(DateTime::from_date_and_time(2107, 12, 31, 23, 59, 58).unwrap()),
518    ///     FileTime::new(159_992_927_980_000_000)
519    /// );
520    /// ```
521    fn from(dt: zip::DateTime) -> Self {
522        assert!(dt.is_valid());
523        let (date, time) = (dt.datepart(), dt.timepart());
524        Self::from_dos_date_time(date, time, None, None)
525            .expect("date and time should be valid as `zip::DateTime`")
526    }
527}
