10impl PackInfo {
11    pub fn write_to<H: Write>(&mut self, header: &mut H) -> std::io::Result<()> {
12        header.write_u8(K_PACK_INFO)?;
13        write_u64(header, self.pos)?;
14        write_u64(header, self.len() as u64)?;
15        header.write_u8(K_SIZE)?;
16        for size in &self.sizes {
17            write_u64(header, *size)?;
18        }
19        header.write_u8(K_CRC)?;
20        let all_crc_defined = self.crcs.iter().all(|f| *f != 0);
21        if all_crc_defined {
22            header.write_u8(1)?; // all defined
23            for crc in self.crcs.iter() {
24                header.write_u32::<LittleEndian>(*crc)?;
25            }
26        } else {
27            header.write_u8(0)?; // not all defined
28            let mut crc_define_bits = BitSet::with_capacity(self.crcs.len());
29
30            for (i, crc) in self.crcs.iter().cloned().enumerate() {
31                if crc != 0 {
32                    crc_define_bits.insert(i);
33                }
34            }
35            let mut temp = Vec::with_capacity(self.len());
36            write_bit_set(&mut temp, &crc_define_bits)?;
37            header.write_all(&temp)?;
38        }
39
40        header.write_u8(K_END)?;
41        Ok(())
42    }
43}
