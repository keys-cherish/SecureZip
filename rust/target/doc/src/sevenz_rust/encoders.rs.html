13pub enum Encoder<W: Write> {
14    LZMA(LZMAWriter<W>),
15    LZMA2(LZMA2Writer<W>),
16    #[cfg(feature = "aes256")]
17    AES(Aes256Sha256Encoder<W>),
18}
19
20impl<W: Write> Write for Encoder<W> {
21    fn write(&mut self, buf: &[u8]) -> std::io::Result<usize> {
22        match self {
23            Encoder::LZMA(w) => w.write(buf),
24            Encoder::LZMA2(w) => w.write(buf),
25            #[cfg(feature = "aes256")]
26            Encoder::AES(w) => w.write(buf),
27        }
28    }
29
30    fn flush(&mut self) -> std::io::Result<()> {
31        match self {
32            Encoder::LZMA(w) => w.flush(),
33            Encoder::LZMA2(w) => w.flush(),
34            #[cfg(feature = "aes256")]
35            Encoder::AES(w) => w.flush(),
36        }
37    }
38}

40pub fn add_encoder<W: Write>(
41    input: CountingWriter<W>,
42    method_config: &SevenZMethodConfiguration,
43) -> Result<Encoder<W>, Error> {
44    let method = method_config.method;
45
46    match method.id() {
47        SevenZMethod::ID_LZMA => {
48            let mut def_opts = LZMA2Options::default();
49            let options = get_lzma2_options(method_config.options.as_ref(), &mut def_opts);
50            let lz = LZMAWriter::new_no_header(input, options, false).map_err(Error::io)?;
51            Ok(Encoder::LZMA(lz))
52        }
53        SevenZMethod::ID_LZMA2 => {
54            let mut def_opts = LZMA2Options::default();
55            let options = get_lzma2_options(method_config.options.as_ref(), &mut def_opts);
56
57            let lz = LZMA2Writer::new(input, options);
58            Ok(Encoder::LZMA2(lz))
59        }
60        #[cfg(feature = "aes256")]
61        SevenZMethod::ID_AES256SHA256 => {
62            let options = match method_config.options.as_ref() {
63                Some(MethodOptions::Aes(p)) => p,
64                _ => return Err(Error::PasswordRequired),
65            };
66
67            Ok(Encoder::AES(Aes256Sha256Encoder::new(input, options)?))
68        }
69        _ => Err(Error::UnsupportedCompressionMethod(
70            method.name().to_string(),
71        )),
72    }
73}

75pub(crate) fn get_options_as_properties<'a>(
76    method: SevenZMethod,
77    options: Option<&MethodOptions>,
78    out: &'a mut [u8],
79) -> &'a [u8] {
80    match method.id() {
81        SevenZMethod::ID_LZMA2 => {
82            let dict_size = options
83                .map(|o| o.get_lzma2_dict_size())
84                .unwrap_or(LZMA2Options::DICT_SIZE_DEFAULT);
85            let lead = dict_size.leading_zeros();
86            let second_bit = (dict_size >> (30u32.wrapping_sub(lead))).wrapping_sub(2);
87            let prop = (19u32.wrapping_sub(lead) * 2 + second_bit) as u8;
88            out[0] = prop;
89            &out[0..1]
90        }
91        SevenZMethod::ID_LZMA => {
92            let mut def_opts = LZMA2Options::default();
93            let options = get_lzma2_options(options, &mut def_opts);
94            let dict_size = options.dict_size;
95            out[0] = options.get_props();
96            out[1..5].copy_from_slice(dict_size.to_le_bytes().as_ref());
97            &out[0..5]
98        }
99        #[cfg(feature = "aes256")]
100        SevenZMethod::ID_AES256SHA256 => {
101            let options = match options.as_ref() {
102                Some(MethodOptions::Aes(p)) => p,
103                _ => return &[],
104            };
105            options.write_properties(out);
106            &out[..34]
107        }
108        _ => &[],
109    }
110}

112#[inline]
113pub(crate) fn get_lzma2_options<'a>(
114    options: Option<&'a MethodOptions>,
115    def_opt: &'a mut LZMA2Options,
116) -> &'a LZMA2Options {
117    let options = match options.as_ref() {
118        Some(MethodOptions::LZMA2(opts)) => opts,
119        Some(MethodOptions::Num(n)) => {
120            def_opt.dict_size = *n;
121            def_opt
122        }
123        _ => {
124            def_opt.dict_size = LZMA2Options::DICT_SIZE_DEFAULT;
125            def_opt
126        }
127    };
128    options
