20    fn get_hash4_size(dict_size: u32) -> u32 {
21        let mut h = dict_size - 1;
22        h |= h >> 1;
23        h |= h >> 2;
24        h |= h >> 4;
25        h |= h >> 8;
26        h >>= 1;
27        h |= 0xffff;
28        if h > (1 << 24) {
29            h >>= 1;
30        }
31        h + 1
32    }
33
34    pub fn get_mem_usage(dict_size: u32) -> u32 {
35        (HASH2_MASK + HASH2_SIZE + Self::get_hash4_size(dict_size)) / (1024 / 4) + 4
36    }

56    pub fn calc_hashes(&mut self, buf: &[u8]) {
57        let tmp = CRC_TABLE[buf[0] as usize] ^ (buf[1] as u32);
58        self.hash2_value = (tmp & HASH2_MASK) as i32;
59        let tmp = tmp ^ ((buf[2] as u32) << 8);
60        self.hash3_value = (tmp & HASH3_MASK) as i32;
61        let tmp = tmp ^ CRC_TABLE[buf[3] as usize] << 5;
62        self.hash4_value = (tmp & self.hash4_mask) as i32;
63    }

78        self.hash3_table[self.hash3_value as usize] = pos;

82    pub fn normalize(&mut self, offset: i32) {
83        LZEncoder::normalize(&mut self.hash2_table, offset);
84        LZEncoder::normalize(&mut self.hash3_table, offset);
85        let hash4_size = self.hash4_size as usize;
86        LZEncoder::normalize(&mut self.hash4_table[..hash4_size], offset);
87    }
